function router5D(data, axisIndex, respoMode) { // Achsen: X=0, Y=1, Z=2, R=3, L=4 const axes = ["X", "Y", "Z", "R", "L"]; // RESPO-Schnur entscheidet den Sprung const jump = { "halten": 0, "nachgeben": 1, "ziehen": 2 }[respoMode] ?? 0; // Achs-Sprung const newAxis = (axisIndex + jump) % 5; return { axis: axes[newAxis], jump, newAxis, data }; }
