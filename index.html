async function loadRespo(name) { // 1) JSON versuchen try { const res = await fetch(`./${name}.json`); if (res.ok) { const data = await res.json(); console.log("RESPO JSON geladen:", name); return data; } } catch(e){} // 2) JS‑Modul versuchen try { const mod = await import(`./${name}.js`); console.log("RESPO JS‑Modul geladen:", name); return mod.default || mod; } catch(e){} // 3) HTML‑Modul versuchen try { const html = await fetch(`./${name}.html`); if (html.ok) { const text = await html.text(); console.log("RESPO HTML geladen:", name); return { html: text }; } } catch(e){} throw new Error("RESPO nicht gefunden: " + name); }
