function interpret(input){ const D = { mech: ["3x6","3x9","PI-3x9","P-3xX"], cmp: ["3x4","3x1","parse3x1"], time: ["TIME-3x11","TIME-3x10","TIME-3x5"], meta: ["3x13","3x12","3x11","3x10"], op: ["AXXX","3x3","3x2","3x1"], rt: ["AXIS","borgProtocol"] }; const parts = input.split("+").map(s=>s.trim()); const found = []; for(const p of parts){ let matched = false; for(const key in D){ if(D[key].includes(p)){ found.push(key); matched = true; } } if(!matched){ return { status:"DEFECT", reason:"UNKNOWN_ELEMENT", element:p, action:"NEUTRALIZE", method:"UNK-PROTO" }; } } const unique = [...new Set(found)]; if(unique.length === 1){ return { status:"OK", discipline:unique[0].toUpperCase(), messID:unique[0].toUpperCase()+"-ID-X", method:unique[0].toUpperCase()+"-PROTO", action:"PROCESS" }; } return { status:"DEFECT", reason:"MIXED_DISCIPLINES", disciplines:unique, action:"SEPARATE", method:"META-PROTO" }; }
