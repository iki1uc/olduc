function parse3x1(r){ const out = { axis:"3x1", core:[], gen:null, tech:null, eich:null, orbit:null, zoom:r.zoom||[4,16,64,128], mode:"" }; if(r.competence){ out.core = r.competence.core; out.gen = r.competence.gen; out.tech = r.competence.tech; out.eich = r.competence.eich; out.orbit = r.orbit || null; out.mode = "NEW"; return out; } if(r.payload){ out.core = r.payload; out.gen = r.id?.gen || null; out.tech = r.id?.tech || null; out.eich = r.id?.eich || null; out.orbit = null; out.mode = "LEGACY"; return out; } out.mode = "EMPTY"; return out; }
